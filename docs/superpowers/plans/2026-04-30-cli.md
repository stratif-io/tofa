# rotp CLI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a full-featured, fully-tested CLI to rotp covering 13 subcommands, shell completions, a man page, and an updated README — while keeping the TUI as the default when no subcommand is given.

**Architecture:** `rotp/src/cli/` replaces the stub `cli.rs`. Each command lives in its own file under `cli/commands/`. `main.rs` dispatches: no subcommand → TUI, otherwise → CLI. Two functions in `rotp-core` are added (`rename_entry`, `uri_to_qr_png`). Tests use `assert_cmd` + `ROTP_PASSPHRASE` env var to avoid interactive prompts.

**Tech Stack:** Rust stable, clap 4 (derive), clap_complete, rpassword 7, arboard 3 (already present), assert_cmd 2, predicates 3, tempfile 3.

```rust
pub mod commands;

use clap::{Parser, Subcommand};
use rpassword::prompt_password;
use rotp_core::store::Vault;
use std::path::PathBuf;

pub type CliResult = Result<(), Box<dyn std::error::Error>>;

#[derive(Parser)]
#[command(
    name = "rotp",
    about = "Eye-candy terminal OTP manager",
    long_about = "rotp manages TOTP codes in an encrypted vault.\n\nRun without arguments to launch the interactive TUI.",
    version,
    arg_required_else_help = false
)]
pub struct Cli {
    /// Path to the vault file (overrides ROTP_VAULT env var)
    #[arg(long, global = true, env = "ROTP_VAULT", value_name = "PATH")]
    pub vault: Option<PathBuf>,

    #[command(subcommand)]
    pub command: Option<Commands>,
}

#[derive(Subcommand)]
pub enum Commands {
    /// Create a new encrypted vault
    Init(commands::init::InitArgs),
    /// Permanently delete the vault
    Destroy,
    /// List all accounts
    List(commands::list::ListArgs),
    /// Show the current TOTP code for an account
    Code(commands::code::CodeArgs),
    /// Add a new account
    Add(commands::add::AddArgs),
    /// Remove an account
    Remove(commands::remove::RemoveArgs),
    /// Rename an account
    Rename(commands::rename::RenameArgs),
    /// Export a QR code for one or all accounts
    Qr(commands::qr::QrArgs),
    /// Change the vault passphrase
    Rekey,
    /// Print shell completions
    Completions(commands::completions::CompletionsArgs),
    /// Export all accounts as plain-text JSON
    Export(commands::export::ExportArgs),
    /// Import accounts from JSON or a migration QR image
    Import(commands::import::ImportArgs),
}

/// Returns the default vault path: ~/.config/rotp/vault.enc
pub fn default_vault_path() -> PathBuf {
    dirs::config_dir()
        .unwrap_or_else(|| PathBuf::from("."))
        .join("rotp")
        .join("vault.enc")
}

/// Resolves the vault path: --vault flag > ROTP_VAULT env > default
pub fn resolve_vault_path(flag: Option<PathBuf>) -> PathBuf {
    flag.unwrap_or_else(default_vault_path)
}

/// Reads the passphrase: ROTP_PASSPHRASE env > interactive prompt
pub fn prompt_passphrase(prompt: &str) -> CliResult {
    // Caller receives the passphrase as a return value, not via this fn.
    // Use `read_passphrase` instead — this fn exists only for the signature.
    let _ = (prompt,);
    Ok(())
}

pub fn read_passphrase(prompt: &str) -> Result<String, Box<dyn std::error::Error>> {
    if let Ok(pass) = std::env::var("ROTP_PASSPHRASE") {
        eprintln!("⚠  Passphrase read from ROTP_PASSPHRASE. Avoid this in production.");
        return Ok(pass);
    }
    let pass = prompt_password(prompt)?;
    Ok(pass)
}

/// Loads vault from path using passphrase. Prints a friendly error if not found.
pub fn open_vault(path: &PathBuf, passphrase: &str) -> Result<Vault, Box<dyn std::error::Error>> {
    if !path.exists() {
        return Err(format!(
            "no vault at {}. Run 'rotp init' to create one.",
            path.display()
        )
        .into());
    }
    let vault = Vault::load(path, passphrase)
        .map_err(|_| "wrong passphrase.")?;
    Ok(vault)
}

/// Case-insensitive substring match. Returns (index, entry) or an error.
pub fn find_entry<'a>(
    vault: &'a Vault,
    name: &str,
) -> Result<(usize, &'a rotp_core::VaultEntry), Box<dyn std::error::Error>> {
    let lower = name.to_lowercase();
    let matches: Vec<(usize, &rotp_core::VaultEntry)> = vault
        .entries()
        .iter()
        .enumerate()
        .filter(|(_, e)| e.name.to_lowercase().contains(&lower))
        .collect();
    match matches.len() {
        0 => Err(format!("no account matching \"{name}\".").into()),
        1 => Ok(matches.into_iter().next().unwrap()),
        _ => {
            let list = matches
                .iter()
                .map(|(_, e)| format!("  {}", e.name))
                .collect::<Vec<_>>()
                .join("\n");
            Err(format!(
                "\"{name}\" matches multiple accounts:\n{list}\nUse a more specific name."
            )
            .into())
        }
    }
}

pub fn dispatch(cmd: Commands, vault_flag: Option<PathBuf>) -> CliResult {
    let vault_path = resolve_vault_path(vault_flag);
    match cmd {
        Commands::Init(args)         => commands::init::run(args, vault_path),
        Commands::Destroy            => commands::destroy::run(vault_path),
        Commands::List(args)         => commands::list::run(args, vault_path),
        Commands::Code(args)         => commands::code::run(args, vault_path),
        Commands::Add(args)          => commands::add::run(args, vault_path),
        Commands::Remove(args)       => commands::remove::run(args, vault_path),
        Commands::Rename(args)       => commands::rename::run(args, vault_path),
        Commands::Qr(args)           => commands::qr::run(args, vault_path),
        Commands::Rekey              => commands::rekey::run(vault_path),
        Commands::Completions(args)  => commands::completions::run(args),
        Commands::Export(args)       => commands::export::run(args, vault_path),
        Commands::Import(args)       => commands::import::run(args, vault_path),
    }
}
```

```rust
use crate::cli::{read_passphrase, CliResult};
use clap::Args;
use rotp_core::store::Vault;
use std::path::PathBuf;

#[derive(Args)]
pub struct InitArgs {}

pub fn run(_args: InitArgs, vault_path: PathBuf) -> CliResult {
    if vault_path.exists() {
        return Err(format!(
            "Vault already exists at {}. Use 'rotp rekey' to change the passphrase.",
            vault_path.display()
        )
        .into());
    }

    let pass = if std::env::var("ROTP_PASSPHRASE").is_ok() {
        read_passphrase("")?
    } else {
        let p = read_passphrase("Passphrase: ")?;
        let confirm = rpassword::prompt_password("Confirm passphrase: ")?;
        if p != confirm {
            return Err("Passphrases do not match.".into());
        }
        p
    };

    let vault = Vault::new();
    vault.save(&vault_path, &pass)?;
    eprintln!("Vault created at {}", vault_path.display());
    Ok(())
}
```

```rust
pub fn run(vault_path: PathBuf) -> CliResult {
```

```rust
use crate::cli::{open_vault, read_passphrase, CliResult};
use clap::Args;
use rotp_core::totp::{generate_code_now, seconds_remaining_now};
use std::path::PathBuf;

#[derive(Args)]
pub struct ListArgs {
    /// Also display current codes and time remaining
    #[arg(long)]
    pub codes: bool,
}

pub fn run(args: ListArgs, vault_path: PathBuf) -> CliResult {
    let pass = read_passphrase("Passphrase: ")?;
    let vault = open_vault(&vault_path, &pass)?;

    if args.codes {
        let secs = seconds_remaining_now();
        for entry in vault.entries() {
```

```rust
use crate::cli::{open_vault, read_passphrase, CliResult};
use clap::Args;
use rotp_core::{
    qr::{parse_input, parse_migration, scan_qr_uri},
    store::{Vault, VaultEntry},
    totp::{generate_code_now, seconds_remaining_now},
};
use std::path::PathBuf;

#[derive(Args)]
pub struct AddArgs {
    /// Account name (required with --secret)
    #[arg(long, value_name = "NAME")]
    pub name: Option<String>,
    /// Base32 TOTP secret
    #[arg(long, value_name = "BASE32")]
    pub secret: Option<String>,
    /// otpauth:// URI
    #[arg(long, value_name = "URI")]
    pub uri: Option<String>,
    /// Path to a QR code image (PNG/JPEG)
    #[arg(long, value_name = "PATH")]
    pub qr: Option<PathBuf>,
}

pub fn run(args: AddArgs, vault_path: PathBuf) -> CliResult {
    let pass = read_passphrase("Passphrase: ")?;
    let mut vault = open_vault(&vault_path, &pass)?;

    if let Some(qr_path) = &args.qr {
        let uri = scan_qr_uri(qr_path)?;
        if uri.starts_with("otpauth-migration://") {
            return import_migration(&uri, &mut vault, &vault_path, &pass);
        }
        let raw = qr_path.to_string_lossy().to_string();
        let otp = parse_input(&raw)?;
        let name = args.name.unwrap_or_else(|| make_name(&otp));
        add_single(&name, &otp.secret, &mut vault, &vault_path, &pass)?;
        return Ok(());
    }

    if let Some(uri) = &args.uri {
        let otp = parse_input(uri)?;
        let name = args.name.unwrap_or_else(|| make_name(&otp));
        add_single(&name, &otp.secret, &mut vault, &vault_path, &pass)?;
        return Ok(());
    }

    if let Some(secret) = &args.secret {
        let name = args.name.ok_or("--name is required when using --secret")?;
        add_single(&name, secret, &mut vault, &vault_path, &pass)?;
        return Ok(());
    }

    Err("Provide --secret, --uri, or --qr.".into())
}

fn make_name(otp: &rotp_core::OtpSecret) -> String {
    match (&otp.meta.issuer, &otp.meta.account) {
        (Some(i), Some(a)) => format!("{i}:{a}"),
        (Some(i), None)    => i.clone(),
        (None, Some(a))    => a.clone(),
        (None, None)       => "unknown".to_string(),
    }
}

fn add_single(name: &str, secret: &str, vault: &mut Vault, path: &PathBuf, pass: &str) -> CliResult {
    let today = chrono::Local::now().format("%Y-%m-%d").to_string();
    vault.add_entry(VaultEntry { name: name.to_string(), secret: secret.to_string(), created_at: today });
    vault.save(path, pass)?;
    let code = generate_code_now(secret).unwrap_or_else(|_| "------".into());
    let secs = seconds_remaining_now();
    println!("Added {name}");
    println!("Current code: {} {}  ({}s)", &code[..3], &code[3..], secs);
    Ok(())
}

fn import_migration(uri: &str, vault: &mut Vault, path: &PathBuf, pass: &str) -> CliResult {
    let accounts = parse_migration(uri)?;
    let count = accounts.len();
    let today = chrono::Local::now().format("%Y-%m-%d").to_string();
    for otp in accounts {
        let name = make_name(&otp);
        vault.add_entry(VaultEntry { name, secret: otp.secret, created_at: today.clone() });
    }
    vault.save(path, pass)?;
    println!("Imported {count} account(s).");
    Ok(())
}
```

```rust
use rotp_core::totp::{generate_code_now, seconds_remaining_now};
use std::io::Write;
use std::path::PathBuf;
use std::time::Duration;

#[derive(Args)]
pub struct CodeArgs {
    /// Account name (partial, case-insensitive)
    pub name: String,
    /// Output bare digits without space (for scripting)
    #[arg(long)]
    pub raw: bool,
    /// Copy code to clipboard
    #[arg(long)]
    pub copy: bool,
    /// Refresh every second until Ctrl+C
    #[arg(long)]
    pub watch: bool,
}

pub fn run(args: CodeArgs, vault_path: PathBuf) -> CliResult {
    let pass = read_passphrase("Passphrase: ")?;
    let vault = open_vault(&vault_path, &pass)?;
    let (_, entry) = find_entry(&vault, &args.name)?;
    let secret = entry.secret.clone();
    let name = entry.name.clone();

    if args.watch {
        return watch_loop(&name, &secret);
    }

    let code = generate_code_now(&secret)?;
    if args.raw {
        println!("{code}");
    } else {
        println!("{} {}", &code[..3], &code[3..]);
    }
    if args.copy {
        match arboard::Clipboard::new().and_then(|mut cb| cb.set_text(code)) {
            Ok(_) => eprintln!("Copied to clipboard."),
            Err(_) => eprintln!("Clipboard unavailable."),
        }
    }
    Ok(())
}

fn watch_loop(name: &str, secret: &str) -> CliResult {
    loop {
        let code = generate_code_now(secret)?;
        let secs = seconds_remaining_now();
        let bar_w = 20usize;
        let filled = ((secs as usize * bar_w) / 30).min(bar_w);
        let bar = format!("{}{}", "█".repeat(filled), "░".repeat(bar_w - filled));
        let line = format!("{name}   {} {}   {bar}   {secs}s", &code[..3], &code[3..]);
        print!("\r\x1b[K{line}");
        std::io::stdout().flush()?;
        std::thread::sleep(Duration::from_secs(1));
    }
}
```

```rust
use crate::cli::{find_entry, open_vault, read_passphrase, CliResult};
use clap::Args;
use rotp_core::{generate_migration_uri, uri_to_qr_lines, uri_to_qr_png};
use std::path::PathBuf;

#[derive(Args)]
pub struct QrArgs {
    /// Account name (partial, case-insensitive)
    pub name: Option<String>,
    /// Export all accounts as a migration QR
    #[arg(long, conflicts_with = "name")]
    pub all: bool,
    /// Save QR as PNG instead of displaying in terminal
    #[arg(long, value_name = "PATH")]
    pub output: Option<PathBuf>,
}

pub fn run(args: QrArgs, vault_path: PathBuf) -> CliResult {
    let pass = read_passphrase("Passphrase: ")?;
    let vault = open_vault(&vault_path, &pass)?;

    let uri = if args.all {
        let accounts: Vec<(&str, &str, &str)> = vault
            .entries()
            .iter()
            .map(|e| (e.name.as_str(), "", e.secret.as_str()))
            .collect();
        generate_migration_uri(&accounts)?
    } else {
        let name = args.name.as_deref().ok_or("provide a name or --all")?;
        let (_, entry) = find_entry(&vault, name)?;
        format!(
            "otpauth://totp/{}?secret={}",
            urlencoding::encode(&entry.name),
            entry.secret
        )
    };

    if let Some(out_path) = args.output {
        uri_to_qr_png(&uri, &out_path)?;
        eprintln!("QR saved to {}", out_path.display());
    } else {
        for line in uri_to_qr_lines(&uri) {
            println!("{line}");
        }
    }
    Ok(())
}
```

```rust
use crate::cli::{open_vault, read_passphrase, CliResult};
use clap::Args;
use rotp_core::{
    qr::{parse_migration, scan_qr_uri},
    store::{Vault, VaultEntry},
};
use std::path::PathBuf;

#[derive(Args)]
pub struct ImportArgs {
    /// JSON file or QR image (PNG/JPEG) to import
    pub file: PathBuf,
}

pub fn run(args: ImportArgs, vault_path: PathBuf) -> CliResult {
    let pass = read_passphrase("Passphrase: ")?;
    let mut vault = open_vault(&vault_path, &pass)?;

    let ext = args.file.extension().and_then(|e| e.to_str()).unwrap_or("").to_lowercase();
    let (imported, skipped) = if ext == "json" {
        import_json(&args.file, &mut vault)?
    } else {
        import_qr(&args.file, &mut vault)?
    };

    vault.save(&vault_path, &pass)?;
    print!("Imported {imported} account(s).");
    if skipped > 0 {
        print!(" Skipped {skipped} duplicate(s).");
    }
    println!();
    Ok(())
}

fn import_json(path: &PathBuf, vault: &mut Vault) -> Result<(usize, usize), Box<dyn std::error::Error>> {
    let content = std::fs::read_to_string(path)?;
    let entries: Vec<VaultEntry> = serde_json::from_str(&content)?;
    let mut imported = 0;
    let mut skipped = 0;
    for entry in entries {
        let dup = vault.entries().iter().any(|e| e.name == entry.name && e.secret == entry.secret);
        if dup {
            skipped += 1;
        } else {
            vault.add_entry(entry);
            imported += 1;
        }
    }
    Ok((imported, skipped))
}

fn import_qr(path: &PathBuf, vault: &mut Vault) -> Result<(usize, usize), Box<dyn std::error::Error>> {
    let uri = scan_qr_uri(path)?;
    let today = chrono::Local::now().format("%Y-%m-%d").to_string();
    if uri.starts_with("otpauth-migration://") {
        let accounts = parse_migration(&uri)?;
        let mut imported = 0;
        let mut skipped = 0;
        for otp in accounts {
            let name = match (&otp.meta.issuer, &otp.meta.account) {
                (Some(i), Some(a)) => format!("{i}:{a}"),
                (Some(i), None) => i.clone(),
                (None, Some(a)) => a.clone(),
                (None, None) => format!("imported-{}", vault.entries().len() + 1),
            };
            let dup = vault.entries().iter().any(|e| e.name == name && e.secret == otp.secret);
            if dup { skipped += 1; } else {
                vault.add_entry(VaultEntry { name, secret: otp.secret, created_at: today.clone() });
                imported += 1;
            }
        }
        Ok((imported, skipped))
    } else {
        Err("QR image is not a migration QR. Use 'rotp add --qr' for single-account QRs.".into())
    }
}
```
